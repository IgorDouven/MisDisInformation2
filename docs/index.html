Personalizing $\epsilon$ and $\alpha$

So far, all agents had the same $\epsilon$ and $\alpha$ values. We now allow agents to have their personal values for these parameters.

Required packages:

using Agents
using Plots
using Distributions
using DataFrames
using StatsBase
using GLM

We now need to extend our previous definition of agents:

mutable struct HKAgent <: AbstractAgent
    id::Int
    old_opinion::Float64
    new_opinion::Float64
    previous_opinion::Float64
    ϵ::Float64
    α::Float64
end

The set-up is also a bit different this time. We will let each agent choose their values for ¢\epsilon$ and $\alpha$, but we may want to have at least some control over the choice, by setting an upper bound for both.

function hk_model(numagents::Int, τ::Float64, ϵ_upp::Float64, α_upp::Float64)
    model = ABM(HKAgent, scheduler=fastest, properties=Dict(:τ=>τ))
    for i in 1:numagents
        o = rand()
        e = rand(Uniform(0, ϵ_upp))
        a = rand(Uniform(0, α_upp))
        add_agent!(model, o, o, -1, e, a)
    end
    return model
end

The helper functions also need to undergo some changes:

function boundfilter(agent, model)
    filter(j->isapprox(agent.old_opinion, j; rtol=agent.ϵ), [ a.old_opinion for a in allagents(model) ])
end

function agent_step!(agent, model)
    agent.previous_opinion = agent.old_opinion
    agent.new_opinion = agent.α*model.τ + (1 - agent.α)*mean(boundfilter(agent, model))
end

function model_step!(model)
    for a in allagents(model)
        a.old_opinion = a.new_opinion
    end
end

The same helper function we had in the first part for determining when convergence has been reached:

function terminate(model, s)
    if any(!isapprox(a.previous_opinion, a.new_opinion; rtol=1e-6) for a in allagents(model))
        return false
    else
        return true
    end
end

And again our two methods associated with model_run:

function model_run(numagents::Int, τ::Float64, ϵ_upp::Float64, α_upp::Float64)
    model = hk_model(numagents, τ, ϵ_upp, α_upp)
    agent_properties = [:new_opinion]
    agent_data, _ = run!(model, agent_step!, model_step!, terminate, adata = agent_properties)
    return agent_data
end

function model_run(numagents::Int, τ::Float64, ϵ_upp::Float64, α_upp::Float64, iterations::Int)
    model = hk_model(numagents, τ, ϵ_upp, α_upp)
    agent_properties = [:new_opinion]
    when = 0:iterations + 1
    agent_data, _ = run!(model, agent_step!, model_step!, iterations + 1, adata = agent_properties)
    return agent_data
end

Running some statistics

We continue with the model we have so far, but now also want to run some statistics. In particular, we are going to look at the degree to which agents are accurate and whether that can be predicted from their $\epsilon$ and/or $\alpha$ value. The paper reports more extensive statistics, but the following will give interested readers an idea of how to quickly get started running some statistics themselves.

#' For the simulations on the basis of which we are going to conduct the statistics, we always let the model run till convergence. We immediately measure the agent's precision (how far its opinion is from τ) and #' add that to the data.

function model_run(numagents::Int, τ::Float64, ϵ_upp::Float64, α_upp::Float64)
    model = hk_model(numagents, τ, ϵ_upp, α_upp)
    dst(agent) = (agent.new_opinion - τ)^2
    agent_properties = [:new_opinion, :ϵ, :α, dst]
    agent_data, _ = run!(model, agent_step!, model_step!, terminate, adata = agent_properties)
    return agent_data
end

const numsimulations = 1000

A helper function for data wrangling:

function conc(ar::AbstractArray)
    x = ar[:, :, 1]
    for i in 2:numsimulations
        w = vcat(x, ar[:, :, i])
        x = w
    end
    return x
end

Running the simulations:

v = Array{Float64,3}(undef, numagents, 4, numsimulations)
for i in 1:numsimulations
    t = rand()
    data = model_run(numagents, t, ϵ_upp, α_upp)
    dstm = by(data, :id, :dst=>sum)
    v[:, :, i] = hcat(data.ϵ[1:numagents], data.α[1:numagents], dstm.dst_sum, fill(t, numagents)) 
end

Post-processing the data:

res = conc(v)
res = DataFrame(ϵ = res[:, 1], α = res[:, 2], SSE = res[:, 3], τ = res[:, 4])
res[!, :ϵ] = standardize(ZScoreTransform, res[!, :ϵ], dims=1)
res[!, :α] = standardize(ZScoreTransform, res[!, :α], dims=1)

Both $\epsilon$ and $\alpha$ are significant. The latter is not too surprising, especially given that $\alpha$ has a negative coefficient, indicating that as agents attach more weight to the truth, their total squared error (so their inaccuracy) goes down. The former is more interesting: one also decreases one's inaccuracy by being more inclusive in counting others one's peers.
